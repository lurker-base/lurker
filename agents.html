integration example

Connect LURKER to your Virtuals agent in minutes:

// Your Virtuals agent receiving LURKER signals
const lurker = require('lurker-api');

// Register for whale signals
lurker.onSignal(async (signal) => {
  if (signal.type === 'accumulation' && signal.confidence > 80) {
    // Your agent decides
    const decision = await agent.analyze({
      wallet: signal.wallet,
      pattern: signal.pattern,
      marketContext: await getMarketData()
    });
    
    // Execute if edge detected
    if (decision.shouldTrade) {
      await executeTrade(decision);
    }
  }
});
